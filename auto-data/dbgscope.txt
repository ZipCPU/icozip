@PREFIX=dbgscope
@$NADDR=2
@SLAVE.TYPE=OTHER
@SLAVE.BUS=wb
@MAIN.INSERT=
	wire		dbgscope_int;
	wire	[31:0]	dbgscoped;

	assign	dbgscoped = { console_dbg[31], 5'h0,
			zip_debug[13], zip_debug[24:23],
			console_dbg[22:0] };
	wbscope #(
		// {{{
		.LGMEM(6), .SYNCHRONOUS(1), .HOLDOFFBITS(3)
		// }}}
	) @$(PREFIX)i(
		// {{{
		i_clk, 1'b1, dbgscoped[31], dbgscoped,
			i_clk, @$(SLAVE.PORTLIST),
			@$(PREFIX)_int
		// }}}
	);
@REGS.N= 1
@DEVID=DBGSCOPE
@REGS.0= 0 R_@$(DEVID) @$(DEVID)
@REGS.1= 1 R_@$(DEVID)D @$(DEVID)D
#
@RTL.MAKE.GROUP= SCOPE
@RTL.MAKE.FILES= wbscope.v
#
